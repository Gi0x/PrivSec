! 1. SafeSearch Enforcements & Redirects

! Brave Search
search.brave.com^$redirect=https://search.brave.com/search?q=$querystring&safe=active

||search.brave.com/*safe=off^all
||search.brave.com/i^all             ! image search unsafe variants
||search.brave.com/v^all             ! video search unsafe variants

! DuckDuckGo (Strict SafeSearch: kp=1)
duckduckgo.com^$redirect=https://duckduckgo.com/?kp=1&q=$querystring

||duckduckgo.com/?kp=0^all
||duckduckgo.com/?ka=off^all
||duckduckgo.com/i^all               ! image search unsafe variants
||duckduckgo.com/v^all               ! video search unsafe variants

! Kagi
kagi.com^$redirect=https://kagi.com/search?q=$querystring&safe=1

||kagi.com/search?safe=0^all
||kagi.com/images^all                 ! image search unsafe variants
||kagi.com/videos^all                 ! video search unsafe variants
