! -------- Brave -------- 
! Block unsafe SafeSearch param

! Hide SafeSearch toggle

search.brave.com##a[href*="safesearch"]

! -------- DuckDuckGo -------- 
! Block unsafe SafeSearch params
||duckduckgo.com/*kp=0*^$important
||duckduckgo.com/*ka=off*^$important

! -------- Kagi --------
! Block unsafe SafeSearch param
||kagi.com/*safe=0*^$important

! Hide SafeSearch UI elements
