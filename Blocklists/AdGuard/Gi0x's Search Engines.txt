! 1. SafeSearch Enforcements & Redirects

! Brave

search.brave.com^$redirect=https://search.brave.com/search?q=$querystring&safe=active
||search.brave.com/*safe=off^$all,important

||search.brave.com/i^$all,important             ! image search unsafe variants
||search.brave.com/v^$all,important             ! video search unsafe variants


! DuckDuckGo (Strict SafeSearch: kp=1)

duckduckgo.com^$redirect=https://duckduckgo.com/?kp=1&q=$querystring
||duckduckgo.com/?kp=0^$all,important
||duckduckgo.com/?ka=off^$all,important

||duckduckgo.com/i^$all,important               ! image search unsafe variants
||duckduckgo.com/v^$all,important               ! video search unsafe variants

! Kagi

kagi.com^$redirect=https://kagi.com/search?q=$querystring&safe=1
||kagi.com/search?safe=0^$all,important

||kagi.com/images^$all,important                 ! image search unsafe variants
||kagi.com/videos^$all,important                 ! video search unsafe variants
