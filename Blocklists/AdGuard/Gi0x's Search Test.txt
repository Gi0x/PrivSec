! Brave (SafeSearch enforced)
||search.brave.com^$important
search.brave.com^$redirect=https://search.brave.com/search?q=$querystring&safe=active
||search.brave.com/*?*safe=off^$all,important
||search.brave.com/i^$document,important       ! block unsafe image search
||search.brave.com/v^$document,important       ! block unsafe video search

! DuckDuckGo (Strict SafeSearch: kp=1)
||duckduckgo.com^$important
duckduckgo.com^$redirect=https://duckduckgo.com/?kp=1&q=$querystring
||duckduckgo.com/*?*kp=0^$all,important
||duckduckgo.com/*?*ka=off^$all,important
||duckduckgo.com/i^$document,important         ! block unsafe image search
||duckduckgo.com/v^$document,important         ! block unsafe video search

! Kagi (SafeSearch)
||kagi.com^$important
kagi.com^$redirect=https://kagi.com/search?q=$querystring&safe=1
||kagi.com/search?*safe=0^$all,important
||kagi.com/images^$document,important          ! block unsafe image search
||kagi.com/videos^$document,important          ! block unsafe video search
