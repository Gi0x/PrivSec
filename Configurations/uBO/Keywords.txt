! EX1: 1 Keyword

! ||*sex*^$all
! ||*ass*^$all
! ||*tit*^$all
! ||*cum*^$all
! ||*anal*^$all

! This matches sex, ass, tit, cum, anal anywhere in a hostname. 
! Use the syntax sparingly because it does not understand word boundaries. 
! It can unintentionally block neutral websites like below: 

! sex > essex.gov.uk, sussex.ac.uk, asexuality.org, unisexfashion.com ,midsexample.net
! ass > passmark.com, constitution.org, entitledto.co.uk
! cum > document, cumulative
! tit > entitled, institution
! anal > analysis

! For better precision and LESS collateral damage, use ||word^$all

! ||sex^$all
! ||ass^$all
! ||tit^$all

! sex > sex.com, sex.net, sex.co.uk
! ass > ass.com, ass.net, ass.co.uk
! tit > tit.com, tit.net, tit.co.uk

! The word must be the entire domain label. 

! ONLY use the syntax below when the word is almost NEVER innocent: 

! ||*poop*^$all
! ||*hentai*^$all
! ||*xxx*^$all
! ||*camgirl*^$all
! ||*onlyfans*^$all

! poop > poop.com, poops.net, poopface.org, poop-rain.xxx, funny-poop-memes.com, superpooper.site
! hentai > hentai.com, hentaigirl.uk, hentai-babes.xyz, 101-hentai-girl.you
! camgirl> camgirls.org, camgirls-101.net, lexy-camgirl.org

! The syntax covers plurals, verbal forms, and compounds. 

! > ||*poop*^$all = “Block anything containing poop anywhere”
! > ||poop^$all = “Block only domains literally called poop”
