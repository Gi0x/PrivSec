# SYNTAX
# –––––––––––––––––––––––––––––––––––––––––––––––––––––––––––––––––––––––––––––––– With ?

# /(^|[-_/])(cunt(bag|s|whore|y)?)([-_/]|$)/i
# /(^|[-_/])(lube(ed|ing|s)?)([-_/]|$)/i

# Matches:
# cunt, cuntbag, cunts, cuntwhore, cunty
# lube, lubed, lubing, lubes

# Use this syntax ONLY if the root word is 'unambiguous'

# –––––––––––––––––––––––––––––––––––––––––––––––––––––––––––––––––––––––––––––––– Without ?

# /(^|[-_/])(ass(es|face|grab|hole))([-_/]|$)/i

# Matches: asses, assface, assgrab, asshole
# Unmatched: ass
# Omit ? because the root word is ambiguous and can form part of other words, ex. assign

# –––––––––––––––––––––––––––––––––––––––––––––––––––––––––––––––––––––––––––––––– Chaining Multiple Terms

# /(^|[-_/])(celeb(s|rity|rities)|gossip(ed|er|ing|s|y)?)([-_/]|$)/i 

# Matches: celebs, celebrity, celebrities, gossip, gossiped, gossiper, gossiping, gossips, gossipy &

# /celebrities/
# site.com/gossiping/news
# celeb-gossip
# news/gossipy_story

# Optional, better to use a single string for 1 term per line

# ––––––––––––––––––––––––––––––––––––––––––––––––––––––––––––––––––––––––––––––––  Multiple Characters w/t 1 Term

# /(^|[-_/])(poop(s|ing|ed|[-_/]?(face|hole|head))?)([-_/]|$)/i

# Matches: 
# poop, poops, pooping, pooped, poopface, poophole, poophead, 
# poop-face, poop-head, poop-hole, poop_face, poop_head, poop_hole, poop/face, poop/head, poop/hole

# ––––––––––––––––––––––––––––––––––––––––––––––––––––––––––––––––––––––––––––––––  [-_/+] w/t multiple Terms

# /(^|[-_/+])((snarp|plonk)(s|ing|ed|[-_/+]?(face|hole))?)([-_/+]|$)/i

# Matches: 
# snarp, snarps, snarping, snarped, snarpface, snarp-hole, snarp_hole, snarp/hole, snarp+hole
# plonk, plonks, plonking, plonked, plonkface, plonk-hole, plonk_hole, plonk/hole, plonk+hole

# [-_/+] = -, _, +  
# Can be expanded to include more symbols, for ex: [-_/+*&%]
